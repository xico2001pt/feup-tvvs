4   /**
5    * A top-down view of a matrix of {@link Square}s.
6    *
7    * @author Jeroen Roosen 
8    */
9   public class Board {
10  
11      /**
12       * The grid of squares with board[x][y] being the square at column x, row y.
13       */
14      private final Square[][] board;
15  
16      /**
17       * Creates a new board.
18       *
19       * @param grid
20       *            The grid of squares with grid[x][y] being the square at column
21       *            x, row y.
22       */
23      Board(Square[][] grid) {
24          assert grid != null;
25          this.board = grid;
26          assert invariant() : "Initial grid cannot contain null squares";
27      }
28  
29      /**
30       * Whatever happens, the squares on the board can't be null.
31       * @return false if any square on the board is null.
32       */
33      protected final boolean invariant() {
34          for (Square[] row : board) {
35              for (Square square : row) {
36                  if (square == null) {
37                      return false;
38                  }
39              }
40          }
41          return true;
42      }
43  
44      /**
45       * Returns the number of columns.
46       *
47       * @return The width of this board.
48       */
49      public int getWidth() {
50          return board.length;
51      }
52  
53      /**
54       * Returns the number of rows.
55       *
56       * @return The height of this board.
57       */
58      public int getHeight() {
59          return board[0].length;
60      }
61  
62      /**
63       * Returns the square at the given <code>x,y</code> position.
64       *
65       * Precondition: The <code>(x, y)</code> coordinates are within the
66       * width and height of the board.
67       *
68       * @param x
69       *            The <code>x</code> position (column) of the requested square.
70       * @param y
71       *            The <code>y</code> position (row) of the requested square.
72       * @return The square at the given <code>x,y</code> position (never null).
73       */
74      public Square squareAt(int x, int y) {
75          assert withinBorders(x, y);
76          Square result = board[x][y];
77          assert result != null : "Follows from invariant.";
78          return result;
79      }
80  
81      /**
82       * Determines whether the given <code>x,y</code> position is on this board.
83       *
84       * @param x
85       *            The <code>x</code> position (row) to test.
86       * @param y
87       *            The <code>y</code> position (column) to test.
88       * @return <code>true</code> iff the position is on this board.
89       */
90      public boolean withinBorders(int x, int y) {
91          return x >= 0 && x < getWidth() && y >= 0 && y < getHeight();
92      }
93  }
