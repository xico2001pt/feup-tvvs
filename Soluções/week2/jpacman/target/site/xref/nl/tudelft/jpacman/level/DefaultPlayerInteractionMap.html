6   /**
7    * An extensible default interaction map for collisions caused by the player.
8    *
9    * The implementation makes use of the interactionmap, and as such can be easily
10   * and declaratively extended when new types of units (ghosts, players, ...) are
11   * added.
12   *
13   * @author Arie van Deursen
14   * @author Jeroen Roosen
15   *
16   */
17  public class DefaultPlayerInteractionMap implements CollisionMap {
18  
19      private final CollisionMap collisions = defaultCollisions();
20  
21      @Override
22      public void collide(Unit mover, Unit movedInto) {
23          collisions.collide(mover, movedInto);
24      }
25  
26      /**
27       * Creates the default collisions Player-Ghost and Player-Pellet.
28       *
29       * @return The collision map containing collisions for Player-Ghost and
30       *         Player-Pellet.
31       */
32      private static CollisionInteractionMap defaultCollisions() {
33          CollisionInteractionMap collisionMap = new CollisionInteractionMap();
34  
35          collisionMap.onCollision(Player.class, Ghost.class,
36              (player, ghost) -> player.setAlive(false));
37  
38          collisionMap.onCollision(Player.class, Pellet.class,
39              (player, pellet) -> {
40                  pellet.leaveSquare();
41                  player.addPoints(pellet.getValue());
42              });
43          return collisionMap;
44      }
45  }
