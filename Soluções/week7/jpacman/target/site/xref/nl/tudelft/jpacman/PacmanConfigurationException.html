3   /**
4    * Exception that is thrown when JPacman cannot be properly loaded
5    * from its resources.
6    *
7    * @author Arie van Deursen, 2014
8    *
9    */
10  public class PacmanConfigurationException extends RuntimeException {
11  
12      /**
13       * A configuration exception with a direct message.
14       *
15       * @param message The exception message.
16       */
17      public PacmanConfigurationException(String message) {
18          super(message);
19      }
20  
21      /**
22       * A configuration exception with a root cause and additional explanation.
23       *
24       * @param message The explanation.
25       * @param cause The root cause.
26       */
27      public PacmanConfigurationException(String message, Throwable cause) {
28          super(message, cause);
29      }
30  }
