19  /**
20   * A level of Pac-Man. A level consists of the board with the players and the
21   * AIs on it.
22   *
23   * @author Jeroen Roosen 
24   */
25  public class Level {
26  
27      /**
28       * The board of this level.
29       */
30      private final Board board;
31  
32      /**
33       * The lock that ensures moves are executed sequential.
34       */
35      private final Object moveLock = new Object();
36  
37      /**
38       * The lock that ensures starting and stopping can't interfere with each
39       * other.
40       */
41      private final Object startStopLock = new Object();
42  
43      /**
44       * The NPCs of this level and, if they are running, their schedules.
45       */
46      private final Map<Ghost, ScheduledExecutorService> npcs;
47  
48      /**
49       * <code>true</code> iff this level is currently in progress, i.e. players
50       * and NPCs can move.
51       */
52      private boolean inProgress;
53  
54      /**
55       * The squares from which players can start this game.
56       */
57      private final List<Square> startSquares;
58  
59      /**
60       * The start current selected starting square.
61       */
62      private int startSquareIndex;
63  
64      /**
65       * The players on this level.
66       */
67      private final List<Player> players;
68  
69      /**
70       * The table of possible collisions between units.
71       */
72      private final CollisionMap collisions;
73  
74      /**
75       * The objects observing this level.
76       */
77      private final Set<LevelObserver> observers;
78  
79      /**
80       * Creates a new level for the board.
81       *
82       * @param board
83       *            The board for the level.
84       * @param ghosts
85       *            The ghosts on the board.
86       * @param startPositions
87       *            The squares on which players start on this board.
88       * @param collisionMap
89       *            The collection of collisions that should be handled.
90       */
91      public Level(Board board, List<Ghost> ghosts, List<Square> startPositions,
92                   CollisionMap collisionMap) {
93          assert board != null;
94          assert ghosts != null;
95          assert startPositions != null;
96  
97          this.board = board;
98          this.inProgress = false;
99          this.npcs = new HashMap<>();
100         for (Ghost ghost : ghosts) {
101             npcs.put(ghost, null);
102         }
103         this.startSquares = startPositions;
104         this.startSquareIndex = 0;
105         this.players = new ArrayList<>();
106         this.collisions = collisionMap;
107         this.observers = new HashSet<>();
108     }
109 
110     /**
111      * Adds an observer that will be notified when the level is won or lost.
112      *
113      * @param observer
114      *            The observer that will be notified.
115      */
116     public void addObserver(LevelObserver observer) {
117         observers.add(observer);
118     }
119 
120     /**
121      * Removes an observer if it was listed.
122      *
123      * @param observer
124      *            The observer to be removed.
125      */
126     public void removeObserver(LevelObserver observer) {
127         observers.remove(observer);
128     }
129 
130     /**
131      * Registers a player on this level, assigning him to a starting position. A
132      * player can only be registered once, registering a player again will have
133      * no effect.
134      *
135      * @param player
136      *            The player to register.
137      */
138     public void registerPlayer(Player player) {
139         assert player != null;
140         assert !startSquares.isEmpty();
141 
142         if (players.contains(player)) {
143             return;
144         }
145         players.add(player);
146         Square square = startSquares.get(startSquareIndex);
147         player.occupy(square);
148         startSquareIndex++;
149         startSquareIndex %= startSquares.size();
150     }
151 
152     /**
153      * Returns the board of this level.
154      *
155      * @return The board of this level.
156      */
157     public Board getBoard() {
158         return board;
159     }
160 
161     /**
162      * Moves the unit into the given direction if possible and handles all
163      * collisions.
164      *
165      * @param unit
166      *            The unit to move.
167      * @param direction
168      *            The direction to move the unit in.
169      */
170     public void move(Unit unit, Direction direction) {
171         assert unit != null;
172         assert direction != null;
173         assert unit.hasSquare();
174 
175         if (!isInProgress()) {
176             return;
177         }
178 
179         synchronized (moveLock) {
180             unit.setDirection(direction);
181             Square location = unit.getSquare();
182             Square destination = location.getSquareAt(direction);
183 
184             if (destination.isAccessibleTo(unit)) {
185                 List<Unit> occupants = destination.getOccupants();
186                 unit.occupy(destination);
187                 for (Unit occupant : occupants) {
188                     collisions.collide(unit, occupant);
189                 }
190             }
191             updateObservers();
192         }
193     }
194 
195     /**
196      * Starts or resumes this level, allowing movement and (re)starting the
197      * NPCs.
198      */
199     public void start() {
200         synchronized (startStopLock) {
201             if (isInProgress()) {
202                 return;
203             }
204             startNPCs();
205             inProgress = true;
206             updateObservers();
207         }
208     }
209 
210     /**
211      * Stops or pauses this level, no longer allowing any movement on the board
212      * and stopping all NPCs.
213      */
214     public void stop() {
215         synchronized (startStopLock) {
216             if (!isInProgress()) {
217                 return;
218             }
219             stopNPCs();
220             inProgress = false;
221         }
222     }
223 
224     /**
225      * Starts all NPC movement scheduling.
226      */
227     private void startNPCs() {
228         for (final Ghost npc : npcs.keySet()) {
229             ScheduledExecutorService service = Executors.newSingleThreadScheduledExecutor();
230 
231             service.schedule(new NpcMoveTask(service, npc),
232                 npc.getInterval() / 2, TimeUnit.MILLISECONDS);
233 
234             npcs.put(npc, service);
235         }
236     }
237 
238     /**
239      * Stops all NPC movement scheduling and interrupts any movements being
240      * executed.
241      */
242     private void stopNPCs() {
243         for (Entry<Ghost, ScheduledExecutorService> entry : npcs.entrySet()) {
244             ScheduledExecutorService schedule = entry.getValue();
245             assert schedule != null;
246             schedule.shutdownNow();
247         }
248     }
249 
250     /**
251      * Returns whether this level is in progress, i.e. whether moves can be made
252      * on the board.
253      *
254      * @return <code>true</code> iff this level is in progress.
255      */
256     public boolean isInProgress() {
257         return inProgress;
258     }
259 
260     /**
261      * Updates the observers about the state of this level.
262      */
263     private void updateObservers() {
264         if (!isAnyPlayerAlive()) {
265             for (LevelObserver observer : observers) {
266                 observer.levelLost();
267             }
268         }
269         if (remainingPellets() == 0) {
270             for (LevelObserver observer : observers) {
271                 observer.levelWon();
272             }
273         }
274     }
275 
276     /**
277      * Returns <code>true</code> iff at least one of the players in this level
278      * is alive.
279      *
280      * @return <code>true</code> if at least one of the registered players is
281      *         alive.
282      */
283     public boolean isAnyPlayerAlive() {
284         for (Player player : players) {
285             if (player.isAlive()) {
286                 return true;
287             }
288         }
289         return false;
290     }
291 
292     /**
293      * Counts the pellets remaining on the board.
294      *
295      * @return The amount of pellets remaining on the board.
296      */
297     public int remainingPellets() {
298         Board board = getBoard();
299         int pellets = 0;
300         for (int x = 0; x < board.getWidth(); x++) {
301             for (int y = 0; y < board.getHeight(); y++) {
302                 for (Unit unit : board.squareAt(x, y).getOccupants()) {
303                     if (unit instanceof Pellet) {
304                         pellets++;
305                     }
306                 }
307             }
308         }
309         assert pellets >= 0;
310         return pellets;
311     }

313     /**
314      * A task that moves an NPC and reschedules itself after it finished.
315      *
316      * @author Jeroen Roosen
317      */
318     private final class NpcMoveTask implements Runnable {
319 
320         /**
321          * The service executing the task.
322          */
323         private final ScheduledExecutorService service;
324 
325         /**
326          * The NPC to move.
327          */
328         private final Ghost npc;
329 
330         /**
331          * Creates a new task.
332          *
333          * @param service
334          *            The service that executes the task.
335          * @param npc
336          *            The NPC to move.
337          */
338         NpcMoveTask(ScheduledExecutorService service, Ghost npc) {
339             this.service = service;
340             this.npc = npc;
341         }
342 
343         @Override
344         public void run() {
345             Direction nextMove = npc.nextMove();
346             if (nextMove != null) {
347                 move(npc, nextMove);
348             }
349             long interval = npc.getInterval();
350             service.schedule(this, interval, TimeUnit.MILLISECONDS);
351         }
352     }

367         /**
368          * The level has been lost. Typically the level should be stopped when
369          * this event is received.

372     }
