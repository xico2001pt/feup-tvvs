1   package nl.tudelft.jpacman.sprite;
2   
3   import java.awt.Graphics;
4   
5   /**
6    * Visual representation of some object.
7    *
8    * @author Jeroen Roosen 
9    */
10  public interface Sprite {
11  
12      /**
13       * Draws the sprite on the provided graphics context.
14       *
15       * @param graphics
16       *            The graphics context to draw.
17       * @param x
18       *            The destination x coordinate to start drawing.
19       * @param y
20       *            The destination y coordinate to start drawing.
21       * @param width
22       *            The width of the destination draw area.
23       * @param height
24       *            The height of the destination draw area.
25       */
26      void draw(Graphics graphics, int x, int y, int width, int height);
27  
28      /**
29       * Returns a portion of this sprite as a new Sprite.
30       *
31       * @param x
32       *            The x start coordinate.
33       * @param y
34       *            The y start coordinate.
35       * @param width
36       *            The width of the target sprite.
37       * @param height
38       *            The height of the target sprite.
39       * @return A new sprite of width x height, or a new {@link EmptySprite} if
40       *         the region was not in the current sprite.
41       */
42      Sprite split(int x, int y, int width, int height);
43  
44      /**
45       * Returns the width of this sprite.
46       *
47       * @return The width in pixels.
48       */
49      int getWidth();
50  
51      /**
52       * Returns the height of this sprite.
53       *
54       * @return The height of this sprite in pixels.
55       */
56      int getHeight();
57  }
