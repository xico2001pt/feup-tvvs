1   package nl.tudelft.jpacman.game;
2   
3   import nl.tudelft.jpacman.level.Level;
4   import nl.tudelft.jpacman.level.PlayerFactory;
5   
6   /**
7    * Factory that provides Game objects.
8    *
9    * @author Jeroen Roosen 
10   */
11  public class GameFactory {
12  
13      /**
14       * The factory providing the player objects.
15       */
16      private final PlayerFactory playerFactory;
17  
18      /**
19       * Creates a new game factory.
20       *
21       * @param playerFactory
22       *            The factory providing the player objects.
23       */
24      public GameFactory(PlayerFactory playerFactory) {
25          this.playerFactory = playerFactory;
26      }
27  
28      /**
29       * Creates a game for a single level with one player.
30       *
31       * @param level
32       *            The level to create a game for.
33       * @return A new single player game.
34       */
35      public Game createSinglePlayerGame(Level level) {
36          return new SinglePlayerGame(playerFactory.createPacMan(), level);
37      }
38  
39      /**
40       * Returns the player factory associated with this game factory.
41       *
42       * @return The player factory associated with this game factory.
43       */
44      protected PlayerFactory getPlayerFactory() {
45          return playerFactory;
46      }
47  }
